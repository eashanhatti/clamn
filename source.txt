module
    Pair : Type -> Type -> Type;
    Pair = record A B where
    	fst : A
    	snd : B
    end;

    sing : Type -> Type;
    sing = fn T => ((import main.Pair)(T, T));

    example : [T : Type] -> T -> ((import main.sing)(T));
    example = fn T x => record [
    	fst = x,
    	snd = x
    ];
end
