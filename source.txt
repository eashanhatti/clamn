module
    Pair : Type -> Type -> Type;
    Pair = record A B where
    	fst : A
    	snd : B
    end;

    example : (import main.Pair)(Fin 1, Fin 1);
    example = record [
    	fst = fin 0,
    	snd = fin 0
    ];

    example2 : [A : Type] -> A -> (import main.Pair)(Fin 1, A);
    example2 = fn A a => record [
    	fst = fin 0,
    	snd = a
    ];
end
